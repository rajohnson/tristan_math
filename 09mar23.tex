    \item Example: 25\% of what number is 6?
        \begin{align*} 
            \text{25\% of } x &= 6 && \text{Let x be the number.}\\[1em]
            \frac{1}{4} \cdot x &= 6 && 25\% = \frac{1}{4}\\[1em]
            4 \cdot \frac{1}{4} \cdot x &= 6 \cdot 4 && \text{The reciprocal of $\frac{1}{4}$ is $\frac{4}{1}$ or 4.}\\[1em]
            x &= 24\\[1em]
        \end{align*}
